ON YOUR LOCAL MACHINE:
───────────────────────────────────────────────────────────
# 1. Create deployment package
./create-deployment-package.sh

# 2. Copy to Hetzner (enter password when prompted)
scp cyber-defense-*.tar.gz root@YOUR_SERVER_IP:/root/


ON YOUR HETZNER SERVER:
───────────────────────────────────────────────────────────
# 3. SSH to server (enter password)
ssh root@YOUR_SERVER_IP

# 4. Install Docker
apt update && apt upgrade -y
curl -fsSL https://get.docker.com | sh
curl -L "https://github.com/docker/compose/releases/latest/download/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose
chmod +x /usr/local/bin/docker-compose

# 5. Setup firewall
apt install -y ufw
ufw allow 22/tcp
ufw allow 8000/tcp
ufw allow 3000/tcp
ufw --force enable

# 6. Extract and start
cd /root
tar -xzf cyber-defense-*.tar.gz
chmod +x *.sh
cp .env.example .env
docker-compose up -d

# 7. Watch it start (wait for "Qwen model ready!")
docker-compose logs -f ollama-init
# Press Ctrl+C when done

# 8. Fix scoring issue (choose option 1 or 4)
./apply-fix.sh

# 9. Test it works
curl http://localhost:8000/health | jq


ACCESS YOUR SYSTEM:
───────────────────────────────────────────────────────────
Dashboard:  http://YOUR_SERVER_IP:3000
Agent API:  http://YOUR_SERVER_IP:8000
API Docs:   http://YOUR_SERVER_IP:8000/docs


USEFUL COMMANDS:
───────────────────────────────────────────────────────────
docker-compose ps              # Check status
docker-compose logs -f         # View logs
docker-compose restart         # Restart all
docker-compose down           # Stop all
docker-compose up -d          # Start all
